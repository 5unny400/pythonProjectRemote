# 通过存储过程删除多表操作

根据自己的业务需求修改where中的条件判断

```sql
-- 删除旧的存储过程
DROP PROCEDURE IF EXISTS delete_multiple_tables;


-- 创建新的存储过程
CREATE PROCEDURE delete_multiple_tables()
BEGIN
    -- 定义变量
    DECLARE run_sql VARCHAR(2000);
    SET @run_sql = CONCAT('DROP TABLE IF EXISTS ',
        (
            SELECT GROUP_CONCAT(table_name) FROM information_schema.tables
            WHERE table_schema = 'test_del_db';
        )
    ,';')
    
    -- 进行判断，如果为空则执行相应操作
    IF @run_sql is null THEN
    -- 空操作...
    SELECT '没有要操作的表';
    ELSE
    -- 非空操作...
    -- 打印下结果
    select @run_sql;
    -- 执行sql
    PREPARE stmt from @run_sql;
    EXECUTE stmt;
    DEALLOCATE PREPARE stmt;
    END IF;

END ;


-- 调用存储过程
call delete_multiple_tables();
```
