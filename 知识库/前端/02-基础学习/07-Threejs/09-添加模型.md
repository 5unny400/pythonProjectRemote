# 添加模型

### STL

```
// 引入STLLoader
import { STLLoader } from 'three/examples/jsm/loaders/STLLoader.js'
```

```
// stlFileList = ['test1.stl', 'test2.stl']
handleSTLData(stlFileList) {
  let that = this
  const loader = new STLLoader();
  stlFileList.forEach(stlFile => {
    loader.load(stlFile, function (geometry) {
      // Phong网格材质 https://threejs.org/docs/index.html#api/zh/materials/MeshPhongMaterial
      const material = new THREE.MeshPhongMaterial({
        // 材质的颜色 -- 随机颜色
        color: '#' + Math.floor(Math.random() * 16777215).toString(16),
        // 开启透明度
        transparent: true,
        // 设置透明度
        opacity: 1,
        // 材质的高光颜色。默认值为0x111111（深灰色）
        specular: 0x111111,
        // 高亮的程度，越高的值越闪亮。默认值为 30。
        shininess: 100,
      });
      // 将几何体渲染为线框
      material.wireframe = true

      // 添加物体
      let mesh = new THREE.Mesh(geometry, material);
      mesh.name = stlFile;
      mesh.castShadow = true;
      // 将模型摆正
      mesh.rotation.x = -0.5 * Math.PI;
      // 缩放 
      mesh.scale.set(0.1, 0.1, 0.1);
      
      that.scene.add(mesh);
    });
  });
}
```
